% El TFG original delega los margenes en geometry:
% \usepackage[a4paper,bindingoffset=3mm,bottom=35mm,includefoot,footskip=12mm]{geometry}
% Por eso este archivo no redefine \textwidth, \oddsidemargin ni
% \topmargin como hacia la plantilla base.

\setlength{\parskip}{0.75em}
\renewcommand{\baselinestretch}{1.25}
\spacing{1}
\raggedbottom

\newcommand{\tfmplainpagestyle}{%
    \fancyhf{}%
    \renewcommand{\headrulewidth}{0pt}%
    \fancyfoot[C]{\thepage}%
}

\newcommand{\tfmmainpagestyle}{%
    \pagestyle{fancy}%
    \fancyhf{}%
    \renewcommand{\headrulewidth}{0.4pt}%
    \renewcommand{\footrulewidth}{0pt}%
    \renewcommand{\chaptermark}[1]{\markboth{Capítulo \thechapter.\ ##1}{}}%
    \renewcommand{\sectionmark}[1]{\markright{\thesection.\ ##1}}%
    \fancyhead[R]{\nouppercase{\rightmark}}%
    \fancyfoot[C]{\thepage}%
}

\fancypagestyle{plain}{\tfmplainpagestyle}
\tfmmainpagestyle
